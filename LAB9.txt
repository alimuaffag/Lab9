CREATE DATABASE ResourceManagementSystem;

# Create ResourceProviders table
CREATE TABLE ResourceProviders (
    id VARCHAR(10) PRIMARY KEY,
    name VARCHAR(20) NOT NULL UNIQUE ,
    email VARCHAR(30) NOT NULL UNIQUE
);

# Create Resources table
CREATE TABLE Resources (
    id VARCHAR(10) PRIMARY KEY,
     name VARCHAR(20) NOT NULL ,
     type VARCHAR(20) NOT NULL ,
    provider_id VARCHAR(10),
FOREIGN KEY (provider_id) REFERENCES ResourceProviders(id)
);

# Create Startups table
CREATE TABLE Startups(
    id VARCHAR(10) PRIMARY KEY,
    name VARCHAR(20) NOT NULL UNIQUE,
    sector VARCHAR(20) NOT NULL,
    email  VARCHAR(30) NOT NULL UNIQUE
);

# Create ResourceRequests table
CREATE TABLE ResourceRequests(
    id VARCHAR(10) PRIMARY KEY,
    startup_id VARCHAR(10),
    resource_id VARCHAR(10),
    description TEXT NOT NULL,
    request_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(10) CHECK ( status= 'Pending' or status= 'Approved' or status= 'Rejected'),
    agreement_percentage DOUBLE NOT NULL,
     FOREIGN KEY (startup_id) REFERENCES Startups(id),
    FOREIGN KEY (resource_id) REFERENCES Resources(id)
);

# Create Rates table
CREATE TABLE Rates(
    id VARCHAR(10) PRIMARY KEY,
    request_id VARCHAR(10),
       comments TEXT NOT NULL,
    rated_at TIMESTAMP NOT NULL,
    FOREIGN KEY (request_id) REFERENCES ResourceRequests(id)
);
